string r="A   3  HIL JM O   2TUVWXY501SE Z  8 ";
char *m_c="AHIMOTUVWXY018";

char mir_c(char c)
{
    if(c>='A'&&c<='Z')
    {
        c=r[c-'A'];
    }
    else if(c>='0'&&c<='9')//如果不是else if 那么变成2的S又会变回S
        c=r[c-'0'+26];
    return c;
}
int  main()
{
    string s;

    int n=s.size();

    for(i=0;i<n/2;i++)
    {
        if(s[i]!=s[n-i-1]) m1=1;
        if(mir_c(s[i])!=s[n-i-1])
