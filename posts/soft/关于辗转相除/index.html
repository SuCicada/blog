#include<iostream>
using namespace std;

int gcd(int a,int b,int& x,int &y){
    if(b==0){
        x = 1;
        y = 0;
        return a;
    }
    int q = gcd(b, a%b, y, x);
    // y 是给b用的
    // a 是大的那个
    y = (q-a*x)/b;
    cout<<a<<" "<<b<<" "<<x<<" "<<y<<endl;
    return q;
}

int main(){
    int x,y;
    int a=12;
    int b=32;
    gcd(a,b,x,y);
}
