十六进制转十进制.h

SuCicada | Suggest Changes
  • soft
#include<stdio.h>
#include<ctype.h>
int sc(char a[])//传进来代表十六进制的字符串数组，返回十进制
{
    int i,n=0,t;
    for(i=0;a[i];i++)
    {
        a[i]=toupper(a[i]);
        if(a[i]>='A')
            t=a[i]-'A'+10;
        else
            t=a[i]-'0';
        n=n*16+t;
    }
    return n;
}
