leetcode - 12. Integer to Roman with Java

方法一，一位一位判断。

class Solution {
   public String intToRoman1(int num) {
        StringBuilder sb = new StringBuilder();
        char ooo[] = new char[]{'I','V','X','L','C','D','M'};

        int wei = 0;
        while(num>0){
            int n = num %10;
            if(n<4){
                for(int i=0;i<n;i++){
                    sb.append(ooo[wei*2]);
                }
            }else if(n==4){
                sb.append(ooo[wei*2+1]);
                sb.append(ooo[wei*2]);
            }else if(n==5){
                sb.append(ooo[wei*2+1]);
            }else if(n>5 && n<9){
                for(int i=0;i<n-5;i++){
                    sb.append(ooo[wei*2]);
                }
                sb.append(ooo[wei*2+1]);
            }else if(n==9){
                sb.append(ooo[wei*2+1+1]);
                sb.append(ooo[wei*2]);
            }
            wei++;
            num = num/10;
        }
        return sb.reverse().toString();
    }
